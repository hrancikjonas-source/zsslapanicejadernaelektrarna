        function init() {
            const grid = document.getElementById('buildings-grid');
            buildings.forEach(b => {
                const card = document.createElement('div');
                card.className = "building-card bg-white p-12 rounded-[3rem] cursor-pointer text-center flex flex-col items-center shadow-lg";
                card.onclick = () => { window.location.hash = b.id; };
                card.innerHTML = `
                    <div class="w-20 h-20 rounded-[1.5rem] ${b.color} text-white flex items-center justify-center mb-8 shadow-xl shadow-slate-100">
                        <i data-lucide="${b.icon}" size="32"></i>
                    </div>
                    <h3 class="text-2xl font-black mb-4 text-slate-800">${b.title}</h3>
                    <p class="text-slate-500 text-sm leading-relaxed">${b.desc}</p>
                    <div class="mt-8 text-blue-600 font-black text-[10px] uppercase tracking-[0.2em] opacity-60 italic">Zobrazit detaily</div>
                `;
                grid.appendChild(card);
            });
            
            renderQuiz();
            lucide.createIcons();
            setTimeout(handleRouting, 100);
        }

        function renderQuiz() {
            const container = document.getElementById('quiz-container');
            container.innerHTML = quizData.map((q, qIdx) => `
                <div class="mb-10 last:mb-0" id="question-${qIdx}">
                    <p class="font-bold text-slate-800 mb-4 flex gap-3 italic">
                        <span class="text-blue-600 not-italic">${qIdx + 1}.</span> ${q.q}
                    </p>
                    <div class="grid gap-3">
                        ${q.options.map((opt, optIdx) => `
                            <label class="quiz-option flex items-center p-4 rounded-xl cursor-pointer" data-q="${qIdx}" data-opt="${optIdx}">
                                <input type="radio" name="q${qIdx}" value="${optIdx}" class="mr-4 w-4 h-4 accent-blue-600">
                                <span class="text-sm text-slate-700 font-medium">${opt}</span>
                            </label>
                        `).join('')}
                    </div>
                </div>
            `).join('');
        }

        function evaluateQuiz() {
            let score = 0;
            quizData.forEach((q, idx) => {
                const selected = document.querySelector(`input[name="q${idx}"]:checked`);
                const optionsElements = document.querySelectorAll(`label[data-q="${idx}"]`);
                
                optionsElements.forEach(label => {
                    const optIdx = parseInt(label.getAttribute('data-opt'));
                    label.classList.add('disabled');
                    label.querySelector('input').disabled = true;

                    if (optIdx === q.correct) {
                        label.classList.add('correct');
                        label.innerHTML += `<i data-lucide="check-circle-2" class="absolute right-4 text-green-600" size="20"></i>`;
                    }
                    if (selected && parseInt(selected.value) === optIdx && optIdx !== q.correct) {
                        label.classList.add('wrong');
                        label.innerHTML += `<i data-lucide="x-circle" class="absolute right-4 text-red-600" size="20"></i>`;
                    }
                });
                if (selected && parseInt(selected.value) === q.correct) score++;
            });

            const result = document.getElementById('quiz-result');
            result.classList.remove('hidden');
            result.className = `mt-8 p-6 rounded-2xl text-center font-bold ${score === quizData.length ? 'bg-green-100 text-green-800' : 'bg-blue-100 text-blue-800'}`;
            result.innerText = `Tvůj výsledek: ${score} z ${quizData.length} správně!`;
            
            document.getElementById('submit-quiz').classList.add('hidden');
            document.getElementById('reset-quiz').classList.remove('hidden');
            lucide.createIcons();
        }

        function resetQuiz() {
            renderQuiz();
            document.getElementById('quiz-result').classList.add('hidden');
            document.getElementById('submit-quiz').classList.remove('hidden');
            document.getElementById('reset-quiz').classList.add('hidden');
            lucide.createIcons();
        }

        function handleRouting() {
            const hash = window.location.hash.split('?')[0].replace('#', '');
            const building = buildings.find(b => b.id === hash);
            if (building) renderDetail(building); else renderHome();
        }

        function renderDetail(b) {
            const content = document.getElementById('detail-content');
            document.getElementById('url-tag').innerText = `LOC_ID: ${b.id.toUpperCase()}`;
            content.innerHTML = `
                <div class="bg-white rounded-[4rem] shadow-2xl overflow-hidden border border-slate-100 fade-in">
                    <div class="${b.color} h-64 flex items-center justify-center relative overflow-hidden">

                            ${b.visual}

                         <div class="relative z-10 p-8 bg-white/10 backdrop-blur-md rounded-[2rem] border border-white/20">
                            <i data-lucide="${b.icon}" size="80" class="text-white"></i>
                         </div>
                    </div>
                    <div class="p-12 md:p-20">
                        <div class="flex flex-col md:flex-row items-center gap-8 mb-12 border-b border-slate-100 pb-12 text-center md:text-left">
                            <div class="p-6 rounded-[2rem] ${b.color} text-white shadow-2xl shadow-slate-200">
                                <i data-lucide="${b.icon}" size="48"></i>
                            </div>

                                <h1 class="text-4xl md:text-6xl font-black text-slate-900 tracking-tighter">${b.title}</h1>

                            </div>
                        </div>
                        <p class="text-xl md:text-2xl text-slate-600 leading-relaxed font-light">${b.content}</p>
                        <button onclick="goBack()" class="mt-20 w-full py-8 bg-slate-900 hover:bg-blue-600 text-white rounded-[2.5rem] font-black uppercase tracking-[0.3em] transition-all shadow-2xl flex items-center justify-center gap-4 text-sm">
                            <i data-lucide="arrow-left" size="20"></i> Zpět na hlavní mapu
                        </button>
                    </div>
                </div>
            `;
            document.getElementById('home-page').classList.add('hidden');
            document.getElementById('detail-page').classList.remove('hidden');
            lucide.createIcons();
            window.scrollTo(0,0);
        }

        function renderHome() {
            document.getElementById('home-page').classList.remove('hidden');
            document.getElementById('detail-page').classList.add('hidden');
            window.scrollTo(0,0);
        }

        function goBack() {
            history.pushState("", document.title, window.location.pathname);
            renderHome();
        }

        window.addEventListener('hashchange', handleRouting);
        window.onload = init;
